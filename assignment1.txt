2. Create a internal hive table "sales_order_csv" which will store csv data sales_order_csv .. make sure to skip header row while creating table
create table sales_order_csv 
(
order_number int,

sales float,

territory string,

)
row format delimited
fields terminated by ','

select product_line, max(order_line_number) as product_with_max_orders from sale_order_orc group by product_line order by product_with_max_orders desc limit 1;

select qtr_id, sum(sales) as total_sales_per_qtr from sale_orc group by qtr_id;

d. In which quarter sales was minimum:
select qtr_id, min(sales) as min_sales_qtr from sales_order_orc group by qtr_id order by min_sales_qtr asc;

e. In which country sales was maximum and in which country sales was minimum:
1. select sales,country from sale_orc order by sales desc limit 1;
2. select sales,country from sale_orc order by sales asc limit 1;
